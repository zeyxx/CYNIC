```javascript
async route(context) {
  // Start episode
  const episode = await this.learningService.startEpisode({
    state: context,
    action: 'route'
  });

  // Route decision
  const result = await this._selectDog(context);

  // WAIT for judgment (NEW)
  const qScore = await this._waitForJudgment(result, context);

  // End episode with REAL reward (NEW)
  await this.learningService.endEpisode(episode.id, qScore);

  return result;
}

async _waitForJudgment(result, context, timeoutMs = 5000) {
  // Check symbiosis cache first (fast path)
  const cached = await this._symbiosisCache.get(context.promptId);
  if (cached && cached.Q) return cached.Q;

  // Wait for judgment event (slow path)
  return new Promise((resolve, reject) => {
    const timeout = setTimeout(() => {
      reject(new Error('Judgment timeout'));
    }, timeoutMs);

    globalEventBus.once('JUDGMENT_CREATED', (judgment) => {
      if (judgment.context?.promptId === context.promptId) {
        clearTimeout(timeout);
        resolve(judgment.Q);
      }
    });
  });
}
```

### Task 4.2: Cron Job Setup

**File**: `packages/node/src/daemon/cron/daily-anchor.js`

```javascript
import cron from 'node-cron';
import { getJudgmentsSince } from '@cynic/persistence';
import { selectAnchorableJudgments, shouldAnchorToday } from '../network/anchor-policy.js';
import { createBlock } from '../../network/block-factory.js';
import { anchorToSolana } from '../../network/solana-anchoring.js';

// Every day at 02:00 UTC (F(8)=21:00 PST-5h)
cron.schedule('0 2 * * *', async () => {
  console.log('[ANCHOR] Daily anchor job started');

  const yesterday = Date.now() - 24 * 60 * 60 * 1000;
  const judgments = await getJudgmentsSince(yesterday);

  if (!shouldAnchorToday(judgments)) {
    console.log('[ANCHOR] No high-stakes judgments. Skipping today.');
    return;
  }

  const toAnchor = selectAnchorableJudgments(judgments);

  try {
    const block = createBlock(toAnchor);
    const result = await anchorToSolana(block, {
      network: 'mainnet',
      burnAmount: 1000 * toAnchor.length  // 1000 $BURN per judgment
    });

    console.log(`[ANCHOR] ‚úì ${toAnchor.length} judgments anchored. Sig: ${result.signature}`);

    // Emit success event
    globalEventBus.publish('BLOCK_ANCHORED', {
      blockHash: block.hash,
      signature: result.signature,
      judgmentCount: toAnchor.length,
      network: 'mainnet'
    });
  } catch (err) {
    console.error('[ANCHOR] ‚úó Anchoring failed:', err);

    // Emit failure event (for monitoring)
    globalEventBus.publish('ANCHOR_FAILED', {
      error: err.message,
      judgmentCount: toAnchor.length
    });
  }
});
```

  const { signature } = req.params;

  // Query Solana RPC to verify anchor
  const onChainData = await solanaClient.getTransaction(signature);

  if (!onChainData) {
    return res.status(404).json({ error: 'Signature not found on-chain' });

  // Parse anchor program logs
  const logs = onChainData.meta.logMessages;
  const rootMatch = logs.find(log => log.includes('Merkle root:'));

    SELECT block_hash FROM block_anchors WHERE signature = $1

  const match = dbAnchor.rows[0]?.block_hash === onChainRoot;

    signature,
    onChainRoot,
    localRoot: dbAnchor.rows[0]?.block_hash,
    match,
    verifiedAt: new Date().toISOString()
